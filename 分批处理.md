```php

$i = 1;         // 从第一页开始
$per = 1000;    // 每次处理条数
$sleep = 10;    // 休息时间
while(true) {

    $list = $users_business->getWechatYanUser([
        'page'      => $i,
        'page_size' => $per,
    ]);

    if (empty($list)) {
        break;
    }

    // 执行业务

    // 还有下一批 休息几秒
    if ((count($list) == $per) && $sleep) {
        sleep($sleep);
    }

    $i++;
}
```
